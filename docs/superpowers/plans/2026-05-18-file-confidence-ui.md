## Task 3: Add File Confidence Snapshot Data

- [ ] **Step 1: Implement snapshot aggregation helpers**

Add helpers near `confidenceByFileEntries()`:

```ts
type WebFileRiskLevel =
  | "missing_ledger"
  | "needs_review"
  | "stale"
  | "superseded"
  | "ok";

interface WebFileConfidenceEntry {
  file: string;
  page_count: number;
  min_score: number | null;
  average_score: number | null;
  risk_level: WebFileRiskLevel;
  flags: string[];
  pages: ConfidenceFilePageSummary[];
}

function webFileConfidenceEntries(vaultDir: string): WebFileConfidenceEntry[] {
  return confidenceByFileEntries(vaultDir, undefined)
    .map((entry) => {
      const scores = entry.pages
        .map((page) => page.score)
        .filter((score): score is number => typeof score === "number");
      const flags = [...new Set(entry.pages.flatMap((page) => page.status.flags))].sort();
      return {
        file: entry.file,
        page_count: entry.pages.length,
        min_score: scores.length === 0 ? null : Math.min(...scores),
        average_score:
          scores.length === 0
            ? null
            : Math.round((scores.reduce((sum, score) => sum + score, 0) / scores.length) * 10000) / 10000,
        risk_level: webFileRiskLevel(flags),
        flags,
        pages: entry.pages,
      };
    })
    .sort(compareWebFileConfidenceEntries);
}
```

Also add `webFileRiskLevel()`, `webFileRiskRank()`, and `compareWebFileConfidenceEntries()`.
